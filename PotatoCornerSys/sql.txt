-- Create Stored Procedures for Stock Management
USE PotatoCorner_DB;
GO

-- Procedure to check if items are in stock
IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND name = 'CheckStockAvailability')
    DROP PROCEDURE CheckStockAvailability;
GO

CREATE PROCEDURE CheckStockAvailability
    @ProductID INT,
    @SizeID INT,
    @FlavorID INT,
    @Quantity INT,
    @IsAvailable BIT OUTPUT,
    @Message NVARCHAR(500) OUTPUT
AS
BEGIN
    DECLARE @SizeStock INT;
    DECLARE @FlavorStock INT;
    DECLARE @FlavorUnitsNeeded INT;
    
    -- Calculate flavor units needed (1 unit serves 10 orders)
    SET @FlavorUnitsNeeded = CEILING(@Quantity / 10.0);
    
    -- Check size stock
    SELECT @SizeStock = StockQuantity
    FROM ProductSizeStock
    WHERE ProductID = @ProductID AND SizeID = @SizeID;
    
    -- Check flavor stock
    SELECT @FlavorStock = StockQuantity
    FROM FlavorStock
    WHERE FlavorID = @FlavorID;
    
    -- Validate stock
    IF @SizeStock IS NULL OR @SizeStock < @Quantity
    BEGIN
        SET @IsAvailable = 0;
        SET @Message = 'This size is currently out of stock.';
        RETURN;
    END
    
    IF @FlavorStock IS NULL OR @FlavorStock < @FlavorUnitsNeeded
    BEGIN
        SET @IsAvailable = 0;
        SET @Message = 'This flavor is currently out of stock.';
        RETURN;
    END
    
    SET @IsAvailable = 1;
    SET @Message = 'Item is available.';
END
GO

-- Procedure to deduct stock after order
IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND name = 'DeductStock')
    DROP PROCEDURE DeductStock;
GO

CREATE PROCEDURE DeductStock
    @ProductID INT,
    @SizeID INT,
    @FlavorID INT,
    @Quantity INT
AS
BEGIN
    DECLARE @FlavorUnitsNeeded INT;
    
    -- Calculate flavor units needed (1 unit serves 10 orders)
    SET @FlavorUnitsNeeded = CEILING(@Quantity / 10.0);
    
    -- Deduct size stock (1:1 ratio)
    UPDATE ProductSizeStock
    SET StockQuantity = StockQuantity - @Quantity,
        LastUpdated = GETDATE()
    WHERE ProductID = @ProductID AND SizeID = @SizeID;
    
    -- Deduct flavor stock (1:10 ratio)
    UPDATE FlavorStock
    SET StockQuantity = StockQuantity - @FlavorUnitsNeeded,
        LastUpdated = GETDATE()
    WHERE FlavorID = @FlavorID;
END
GO

-- Procedure to get available sizes for a product
IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND name = 'GetAvailableSizes')
    DROP PROCEDURE GetAvailableSizes;
GO

CREATE PROCEDURE GetAvailableSizes
    @ProductID INT
AS
BEGIN
    SELECT 
        ps.SizeID,
        ps.SizeName,
        pss.StockQuantity
    FROM ProductSizes ps
    INNER JOIN ProductSizeStock pss ON ps.SizeID = pss.SizeID
    WHERE pss.ProductID = @ProductID
        AND pss.StockQuantity > 0
    ORDER BY ps.SizeID;
END
GO

-- Procedure to get available flavors
IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND name = 'GetAvailableFlavors')
    DROP PROCEDURE GetAvailableFlavors;
GO

CREATE PROCEDURE GetAvailableFlavors
AS
BEGIN
    SELECT 
        pf.FlavorID,
        pf.FlavorName,
        fs.StockQuantity
    FROM ProductFlavors pf
    INNER JOIN FlavorStock fs ON pf.FlavorID = fs.FlavorID
    WHERE fs.StockQuantity > 0
    ORDER BY pf.FlavorID;
END
GO

DECLARE @ConstraintName NVARCHAR(200)
DECLARE @SQL NVARCHAR(500)

PRINT 'Finding all constraints on Orders table...';

DECLARE constraint_cursor CURSOR FOR
SELECT name
FROM sys.check_constraints
WHERE parent_object_id = OBJECT_ID('Orders')

OPEN constraint_cursor
FETCH NEXT FROM constraint_cursor INTO @ConstraintName

WHILE @@FETCH_STATUS = 0
BEGIN
    PRINT 'Dropping constraint: ' + @ConstraintName;
    SET @SQL = 'ALTER TABLE Orders DROP CONSTRAINT [' + @ConstraintName + ']';
    EXEC sp_executesql @SQL;
    PRINT 'Dropped: ' + @ConstraintName;
    FETCH NEXT FROM constraint_cursor INTO @ConstraintName
END

CLOSE constraint_cursor
DEALLOCATE constraint_cursor

PRINT 'All old constraints removed!';
